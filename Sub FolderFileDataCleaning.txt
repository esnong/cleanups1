Latest Headcount Total = 
VAR LatestYearMonth =
    CALCULATETABLE (
        TOPN (
            1,
            SUMMARIZE (
                'STA Cost Center',
                YEAR('STA Cost Center'[Date]),
                MONTH('STA Cost Center'[Date])
            ),
            YEAR('STA Cost Center'[Date]) * 100 + MONTH('STA Cost Center'[Date]),
            DESC
        ),
        ALL('STA Cost Center')
    )
RETURN
    CALCULATE (
        SUM('STA Cost Center'[Headcount]),
        FILTER (
            'STA Cost Center',
            YEAR('STA Cost Center'[Date]) = SELECTCOLUMNS(LatestYearMonth, "Year", YEAR('STA Cost Center'[Date])) &&
            MONTH('STA Cost Center'[Date]) = SELECTCOLUMNS(LatestYearMonth, "Month", MONTH('STA Cost Center'[Date]))
        )
    )
