Total Headcount Latest Month =
VAR LatestYearMonth =
    CALCULATETABLE(
        TOPN(1,
            SUMMARIZE(
                'STA Cost Center',
                YEAR('STA Cost Center'[Date]),
                MONTH('STA Cost Center'[Date])
            ),
            YEAR('STA Cost Center'[Date]) * 100 + MONTH('STA Cost Center'[Date]),
            DESC
        ),
        ALL('STA Cost Center')
    )
VAR LatestYear = MAXX(LatestYearMonth, YEAR('STA Cost Center'[Date]))
VAR LatestMonth = MAXX(LatestYearMonth, MONTH('STA Cost Center'[Date]))

RETURN
CALCULATE(
    SUM('STA Cost Center'[Headcount]),
    YEAR('STA Cost Center'[Date]) = LatestYear,
    MONTH('STA Cost Center'[Date]) = LatestMonth
)
