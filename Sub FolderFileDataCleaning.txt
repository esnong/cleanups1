Total Headcount Latest Month =
VAR YearMonthTable =
    ADDCOLUMNS(
        SUMMARIZE('STA Cost Center', 'STA Cost Center'[Date]),
        "Year", YEAR('STA Cost Center'[Date]),
        "Month", MONTH('STA Cost Center'[Date])
    )

VAR LatestYearMonth =
    TOPN(
        1,
        YearMonthTable,
        [Year] * 100 + [Month],
        DESC
    )

VAR LatestYear = MAXX(LatestYearMonth, [Year])
VAR LatestMonth = MAXX(LatestYearMonth, [Month])

RETURN
CALCULATE(
    SUM('STA Cost Center'[Headcount]),
    YEAR('STA Cost Center'[Date]) = LatestYear,
    MONTH('STA Cost Center'[Date]) = LatestMonth
)
