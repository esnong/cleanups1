Latest Headcount Total = 
VAR LatestYearMonth =
    CALCULATETABLE(
        TOPN(
            1,
            ADDCOLUMNS(
                SUMMARIZE('STA Cost Center', 
                    "Year", YEAR('STA Cost Center'[Date]), 
                    "Month", MONTH('STA Cost Center'[Date])
                ),
                "YearMonthNumber", [Year] * 100 + [Month]
            ),
            [YearMonthNumber], DESC
        ),
        ALL('STA Cost Center')
    )

VAR SelectedYear = SELECTCOLUMNS(LatestYearMonth, "Year", [Year])
VAR SelectedMonth = SELECTCOLUMNS(LatestYearMonth, "Month", [Month])

RETURN
    CALCULATE(
        SUM('STA Cost Center'[Headcount]),
        FILTER(
            'STA Cost Center',
            YEAR('STA Cost Center'[Date]) = SelectedYear &&
            MONTH('STA Cost Center'[Date]) = SelectedMonth
        )
    )
