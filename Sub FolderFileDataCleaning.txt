Headcount (Latest or Selected Year) :=
VAR SelectedYear = SELECTEDVALUE('General - Years'[Year])
VAR LatestYear =
    CALCULATE(
        MAX(YEAR('EDB - Manpower'[Updated as of])),
        ALL('EDB - Manpower')
    )
VAR TargetYear = IF(ISBLANK(SelectedYear), LatestYear, SelectedYear)
RETURN
CALCULATE(
    DISTINCTCOUNT('EDB - Manpower'[Name of Personnel]),
    YEAR('EDB - Manpower'[Updated as of]) = TargetYear
)
