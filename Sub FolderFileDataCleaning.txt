Latest Headcount Total = 
VAR SummaryTable =
    ADDCOLUMNS(
        SUMMARIZE(
            'STA Cost Center',
            "Year", YEAR('STA Cost Center'[Date]),
            "Month", MONTH('STA Cost Center'[Date])
        ),
        "YearMonthNumber", [Year] * 100 + [Month]
    )

VAR LatestPeriod =
    TOPN(1, SummaryTable, [YearMonthNumber], DESC)

VAR LatestYear = MAXX(LatestPeriod, [Year])
VAR LatestMonth = MAXX(LatestPeriod, [Month])

RETURN
    CALCULATE(
        SUM('STA Cost Center'[Headcount]),
        FILTER(
            'STA Cost Center',
            YEAR('STA Cost Center'[Date]) = LatestYear &&
            MONTH('STA Cost Center'[Date]) = LatestMonth
        )
    )
