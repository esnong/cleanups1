AmountUSD_to_SGD :=
SUMX (
    'EDB - Expenses',
    VAR ThisYear = RELATED('General - Years'[Year])
    VAR Rate =
        CALCULATE (
            MAX('General - Ex Rate(USD > SGD)'[RateToSGD]),
            'General - Ex Rate(USD > SGD)'[Currency] = "USD",
            'General - Ex Rate(USD > SGD)'[Year] = ThisYear
        )
    RETURN 'EDB - Expenses'[AmountUSD] * Rate
)
