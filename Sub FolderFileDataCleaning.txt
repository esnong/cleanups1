Function GetMonthAbbreviation(monthNumber As String) As String
    Select Case monthNumber
        Case "01"
            GetMonthAbbreviation = "Jan"
        Case "02"
            GetMonthAbbreviation = "Feb"
        Case "03"
            GetMonthAbbreviation = "Mar"
        Case "04"
            GetMonthAbbreviation = "Apr"
        Case "05"
            GetMonthAbbreviation = "May"
        Case "06"
            GetMonthAbbreviation = "Jun"
        Case "07"
            GetMonthAbbreviation = "Jul"
        Case "08"
            GetMonthAbbreviation = "Aug"
        Case "09"
            GetMonthAbbreviation = "Sep"
        Case "10"
            GetMonthAbbreviation = "Oct"
        Case "11"
            GetMonthAbbreviation = "Nov"
        Case "12"
            GetMonthAbbreviation = "Dec"
        Case Else
            GetMonthAbbreviation = "Invalid"
    End Select
End Function
