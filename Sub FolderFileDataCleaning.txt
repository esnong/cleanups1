Latest Headcount =
CALCULATE(
    MAX('STA Cost Center'[Headcount]),
    FILTER(
        'STA Cost Center',
        NOT(ISBLANK('STA Cost Center'[Headcount]))
            && 'STA Cost Center'[Date] = 
                CALCULATE(
                    MAX('STA Cost Center'[Date]),
                    FILTER('STA Cost Center', NOT(ISBLANK('STA Cost Center'[Headcount])))
                )
    )
)
