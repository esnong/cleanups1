Headcount Latest or Selected Year :=
VAR SelectedYear = SELECTEDVALUE('Date'[Year])
VAR LatestYear =
    CALCULATE(
        MAX('Date'[Year]),
        ALL('Date')
    )
VAR TargetYear = IF(ISBLANK(SelectedYear), LatestYear, SelectedYear)
RETURN
CALCULATE(
    [Headcount],  -- Replace this with your actual headcount measure or expression
    'Date'[Year] = TargetYear
)
